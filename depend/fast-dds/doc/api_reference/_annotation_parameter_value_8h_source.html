584  std::string to_string() const
585  {

594  case TK_UINT16:

596  case TK_INT32:
597  return std::to_string(m_int32_value);
598  case TK_UINT32:
599  return std::to_string(m_uint32_value);
600  case TK_INT64:
601  return std::to_string(m_int64_value);

603  return std::to_string(m_uint64_value);
604  case TK_FLOAT32:
605  return std::to_string(m_float32_value);
606  case TK_FLOAT64:
607  return std::to_string(m_float64_value);
608  case TK_FLOAT128:
609  return std::to_string(m_float128_value);
610  case TK_CHAR8:
611  return std::to_string(m_char_value);
612  case TK_CHAR16:
613  return std::to_string(m_wchar_value);
614  case TK_ENUM:
615  return std::to_string(m_enumerated_value);
616  case TK_STRING16:
617  {
618  std::wstring_convert<std::codecvt_utf8<wchar_t>> conv;
619  return conv.to_bytes(m_string16_value);
620  }
621  case TK_STRING8:
622  case TK_NONE: // Cheat!
623  return m_string8_value;
624  default:
625  return "";
626  }

632  void from_string(const std::string& value)
633  {
634  switch(m__d)
635  {
636  case TK_BOOLEAN:
637  {
638  std::string val_ = value;
639  std::transform(val_.begin(), val_.end(), val_.begin(),
640  [](unsigned char c){ return static_cast<char>(std::tolower(c));});
641  boolean_value(val_.compare("0") != 0 || val_.compare(CONST_TRUE) == 0);
642  }
643  break;
644  case TK_BYTE:
645  {
646  byte_value(static_cast<uint8_t>(std::stoul(value)));
647  }
648  break;
649  case TK_INT16:
650  {
651  int16_value(static_cast<int16_t>(std::stoi(value)));
652  }
653  break;
654  case TK_INT32:
655  {
656  int32_value(static_cast<int32_t>(std::stoi(value)));
657  }
658  break;
659  case TK_INT64:
660  {
661  int64_value(static_cast<int64_t>(std::stoll(value)));
662  }
663  break;
664  case TK_UINT16:
665  {
666  uint_16_value(static_cast<uint16_t>(std::stoul(value)));
667  }
668  break;
669  case TK_UINT32:
670  {
671  uint32_value(static_cast<uint32_t>(std::stoul(value)));
672  }
673  break;
674  case TK_UINT64:
675  {
676  uint64_value(static_cast<uint64_t>(std::stoull(value)));
677  }
678  break;
679  case TK_FLOAT32:
680  {
681  float32_value(std::stof(value));
682  }
683  break;
684  case TK_FLOAT64:
685  {
686  float64_value(std::stod(value));
687  }
688  break;
689  case TK_FLOAT128:
690  {
691  float128_value(std::stold(value));
692  }
693  break;
694  case TK_CHAR8:
695  {
696  char_value(value.c_str()[0]);
697  }
698  break;
699  case TK_CHAR16:
700  {
701  std::wstring_convert<std::codecvt_utf8<wchar_t>> conv;
702  wchar_value(conv.from_bytes(value).c_str()[0]);
703  }
704  break;
705  case TK_STRING8:
706  case TK_NONE: // Cheat!
707  {
708  string8_value(value);
709  }
710  break;
711  case TK_STRING16:
712  {
713  std::wstring_convert<std::codecvt_utf8<wchar_t>> conv;
714  string16_value(conv.from_bytes(value));
715  }
716  break;
717  case TK_ENUM:
718  {
719  // TODO Translate from enum value name to integer value
720  enumerated_value(static_cast<uint32_t>(std::stoul(value)));
721  }
722  break;
723  default:
724  break;
725  }
726  }
