476  bool ensure_capacity()
477  {
478  size_type size = collection_.size();
479  size_type cap = collection_.capacity();
480  if (size == cap)
481  {
482  // collection is full, check resource limit
483  if (cap < configuration_.maximum)
484  {
485  // increase collection capacity
486  assert(configuration_.increment > 0);
487  cap += configuration_.increment;
488  cap = std::min(cap, configuration_.maximum);
489  collection_.reserve(cap);
490  }
491  else
492  {
493  return false;
494  }
495  }
496 
497  return true;
498  }

509  template <typename Enabler = _KeepOrderEnabler>
510  typename std::enable_if<!Enabler::value, void>::type do_remove(
511  iterator it)
512  {
513  // Copy last element into the element being removed
514  if (it != --collection_.end())
515  {
516  *it = std::move(collection_.back());
517  }
518 
519  // Then drop last element
520  collection_.pop_back();
521  }
522 
533  template <typename Enabler = _KeepOrderEnabler>
534  typename std::enable_if<Enabler::value, void>::type do_remove(
535  iterator it)
536  {
537  collection_.erase(it);
538  }
