121  void base_update(
122  T base) noexcept
123  {
124  // Do nothing if base is not changing
125  if (base == base_)
126  {
127  return;
128  }
129 
130  Diff d_func;
131  if (base > base_)
132  {
133  // Current content should move left
134  uint32_t n_bits = d_func(base, base_);
135  shift_map_left(n_bits);
136  }
137  else
138  {
139  // Current content should move right
140  uint32_t n_bits = d_func(base_, base);
141  shift_map_right(n_bits);
142  }
143 
144  // Update base and range
145  base_ = base;
146  range_max_ = base_ + (NBITS - 1);
147  }

174  T min() const noexcept
175  {
176  // Traverse through the significant items on the bitmap
177  T item = base_;
178  uint32_t n_longs = (num_bits_ + 31UL) / 32UL;
179  for (uint32_t i = 0; i < n_longs; i++)
180  {
181  // Check if item has at least one bit set
182  uint32_t bits = bitmap_[i];
183  if (bits)
184  {
185  // We use an intrinsic to find the index of the highest bit set.
186  // Most modern CPUs have an instruction to count the leading zeroes of a word.
187  // The number of leading zeroes will give us the index we need.
188 #if _MSC_VER
189  unsigned long bit;
190  _BitScanReverse(&bit, bits);
191  uint32_t offset = 31UL ^ bit;
192 #else
193  uint32_t offset = __builtin_clz(bits);
194 #endif
195 
196  // Found first bit set in bitmap
197  return item + offset;
198  }
199 
200  // There are 32 items on each bitmap item.
201  item = item + 32UL;
202  }
203 
204  return base_;
205  }

214  bool is_set(
215  const T& item) const noexcept
216  {
217  // Check item is inside the allowed range.
218  if ((item >= base_) && (range_max_ >= item))
219  {
220  // Calc distance from base to item, and check the corresponding bit.
221  Diff d_func;
222  uint32_t diff = d_func(item, base_);
223  if (diff < num_bits_)
224  {
225  uint32_t pos = diff >> 5;
226  diff &= 31UL;
227  return (bitmap_[pos] & (1UL << (31UL - diff))) != 0;
228  }
229  }
230 
231  return false;
232  }

270  void add_range(
271  const T& from,
272  const T& to)
273  {
274  constexpr uint32_t full_mask = 0xFFFFFFFF;
275 
276  // Adapt incoming range to range limits
277  T min = (base_ >= from) ? base_ : from;
278  T max = (to >= base_ + NBITS) ? base_ + NBITS : to;
279 
280  // Check precondition. Max should be explicitly above min.
281  if (min >= max)
282  {
283  return;
284  }
285 
286  // Calc offset (distance from base) and num_bits (bits to be set)
287  Diff d_func;
288  uint32_t offset = d_func(min, base_); // Bit position from base
289  uint32_t n_bits = d_func(max, min); // Number of bits pending
290 
291  num_bits_ = std::max(num_bits_, offset + n_bits);
292 
293  uint32_t pos = offset >> 5; // Item position
294  offset &= 31UL; // Bit position inside item
295  uint32_t mask = full_mask; // Mask with all bits set
296  mask >>= offset; // Remove first 'offset' bits from mask
297  uint32_t bits_in_mask = 32UL - offset; // Take note of number of set bits in mask
298 
299  // This loop enters whenever the whole mask should be added
300  while (n_bits >= bits_in_mask)
301  {
302  bitmap_[pos] |= mask; // Set whole mask of bits
303  pos++; // Go to next position in the array
304  n_bits -= bits_in_mask; // Decrease number of pending bits
305  mask = full_mask; // Mask with all bits set
306  bits_in_mask = 32UL; // All bits set in mask (32)
307  }
308 
309  // This condition will be true if the last bits of the mask should not be used
310  if (n_bits > 0)
311  {
312  bitmap_[pos] |= mask & (full_mask << (bits_in_mask - n_bits));
313  }
314  }

323  const T& item) noexcept
324  {
325  // Check item is inside the allowed range.

327  if ((item >= base_) && (max_value >= item))

385  template<class UnaryFunc>
386  void for_each(
387  UnaryFunc f) const
388  {
389  T item = base_;
390 
391  // Traverse through the significant items on the bitmap
392  uint32_t n_longs = (num_bits_ + 31UL) / 32UL;
393  for (uint32_t i = 0; i < n_longs; i++)
394  {
395  // Traverse through the bits set on the item, msb first.
396  // Loop will stop when there are no bits set.
397  uint32_t bits = bitmap_[i];
398  while(bits)
399  {
400  // We use an intrinsic to find the index of the highest bit set.
401  // Most modern CPUs have an instruction to count the leading zeroes of a word.
402  // The number of leading zeroes will give us the index we need.
403 #if _MSC_VER
404  unsigned long bit;
405  _BitScanReverse(&bit, bits);
406  uint32_t offset = 31UL ^ bit;
407 #else
408  uint32_t offset = __builtin_clz(bits);
409  uint32_t bit = 31UL ^ offset;
410 #endif
411 
412  // Call the function for the corresponding item
413  f(item + offset);
414 
415  // Clear the most significant bit
416  bits &= ~(1UL << bit);
417  }
418 
419  // There are 32 items on each bitmap item.
420  item = item + 32UL;
421  }
422  }

432  void shift_map_left(
433  uint32_t n_bits)
434  {
435  if (n_bits >= num_bits_)
436  {
437  // Shifting more than most significant. Clear whole bitmap.
438  num_bits_ = 0;
439  bitmap_.fill(0UL);
440  }
441  else
442  {
443  // Significant bit will move left by n_bits
444  num_bits_ -= n_bits;
445 
446  // Div and mod by 32
447  uint32_t n_items = n_bits >> 5;
448  n_bits &= 31UL;
449  if (n_bits == 0)
450  {
451  // Shifting a multiple of 32 bits, just move the bitmap integers
452  std::copy(bitmap_.begin() + n_items, bitmap_.end(), bitmap_.begin());
453  std::fill_n(bitmap_.rbegin(), n_items, 0);
454  }
455  else
456  {
457  // Example. Shifting 44 bits. Should shift one complete word and 12 bits.
458  // Need to iterate forward and take 12 bits from next word (shifting it 20 bits).
459  // aaaaaaaa bbbbbbbb cccccccc dddddddd
460  // bbbbbccc bbbbbbbb cccccccc dddddddd
461  // bbbbbccc cccccddd ddddd000 dddddddd
462  // bbbbbccc cccccddd ddddd000 00000000
463  uint32_t overflow_bits = 32UL - n_bits;
464  size_t last_index = NITEMS - 1u;
465  for (size_t i = 0, n = n_items; n < last_index; i++, n++)
466  {
467  bitmap_[i] = (bitmap_[n] << n_bits) | (bitmap_[n + 1] >> overflow_bits);
468  }
469  // Last one does not have next word
470  bitmap_[last_index - n_items] = bitmap_[last_index] << n_bits;
471  // Last n_items will become 0
472  std::fill_n(bitmap_.rbegin(), n_items, 0);
473  }
474  }
475  }
476 
477  void shift_map_right(
478  uint32_t n_bits)
479  {
480  if (n_bits >= NBITS)
481  {
482  // Shifting more than total bitmap size. Clear whole bitmap.
483  num_bits_ = 0;
484  bitmap_.fill(0UL);
485  }
486  else
487  {
488  // Detect if highest bit will be dropped and take note, as we will need
489  // to find new maximum bit in that case
490  uint32_t new_num_bits = num_bits_ + n_bits;
491  bool find_new_max = new_num_bits > NBITS;
492 
493  // Div and mod by 32
494  uint32_t n_items = n_bits >> 5;
495  n_bits &= 31UL;
496  if (n_bits == 0)
497  {
498  // Shifting a multiple of 32 bits, just move the bitmap integers
499  std::copy(bitmap_.rbegin() + n_items, bitmap_.rend(), bitmap_.rbegin());
500  std::fill_n(bitmap_.begin(), n_items, 0);
501  }
502  else
503  {
504  // Example. Shifting 44 bits. Should shift one complete word and 12 bits.
505  // Need to iterate backwards and take 12 bits from previous word (shifting it 20 bits).
506  // aaaaaaaa bbbbbbbb cccccccc dddddddd
507  // aaaaaaaa bbbbbbbb cccccccc bbbccccc
508  // aaaaaaaa bbbbbbbb aaabbbbb bbbccccc
509  // aaaaaaaa 000aaaaa aaabbbbb bbbccccc
510  // 00000000 000aaaaa aaabbbbb bbbccccc
511  uint32_t overflow_bits = 32UL - n_bits;
512  size_t last_index = NITEMS - 1u;
513  for (size_t i = last_index, n = last_index - n_items; n > 0; i--, n--)
514  {
515  bitmap_[i] = (bitmap_[n] >> n_bits) | (bitmap_[n - 1] << overflow_bits);
516  }
517  // First item does not have previous word
518  bitmap_[n_items] = bitmap_[0] >> n_bits;
519  // First n_items will become 0
520  std::fill_n(bitmap_.begin(), n_items, 0);
521  }
522 
523  num_bits_ = new_num_bits;
524  if (find_new_max)
525  {
526  calc_maximum_bit_set(NITEMS, n_items);
527  }
528  }
529  }

531  void calc_maximum_bit_set(
532  uint32_t starting_index,
533  uint32_t min_index)
534  {
535  num_bits_ = 0;
536  for (uint32_t i = starting_index; i > min_index;)
537  {
538  --i;
539  uint32_t bits = bitmap_[i];
540  if (bits != 0)
541  {
542  bits = (bits & ~(bits - 1));
543 #if _MSC_VER
544  unsigned long bit;
545  _BitScanReverse(&bit, bits);
546  uint32_t offset = (31UL ^ bit) + 1;
547 #else
548  uint32_t offset = __builtin_clz(bits) + 1;
549 #endif
550  num_bits_ = (i << 5UL) + offset;
551  break;
552  }
553  }
554  }
